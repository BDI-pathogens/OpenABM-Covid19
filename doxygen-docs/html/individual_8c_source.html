55 /*****************************************************************************************
56 * Name: initialize_hazard
57 * Description: gives each individual an initial amount of hazard which they burn through
58 * with each interaction. The value is adjusted by their relative susceptibility
59 * per interaction.
60 * Returns: void
61 ******************************************************************************************/
62 void initialize_hazard(
63  individual *indiv,
64  parameters *params
65 )
66 {
67  double rate = 1.0;
68  if( indiv->age_group == AGE_0_17 )
69  rate /= params->adjusted_susceptibility_child;
70  if( indiv->age_group == AGE_65 )
71  rate /= params->adjusted_susceptibility_elderly;
72 
73  indiv->hazard = rate * gsl_ran_exponential( rng, 1.0 );
74 }

104 /*****************************************************************************************
105 * Name: set_age_group
106 * Description: sets a person's age group and draws other properties based up on this
107 *
108 * 1. The number of random interactions the person has a day which is
109 * drawn from a negative binomial with an age dependent mean
110 * 2. Which work network they are a member of - some adults are
111 * assigned to the child/elderly network (i.e. teacher/carers)
112 *
113 * Returns: void
114 ******************************************************************************************/
115 void set_age_group( individual *indiv, parameters *params, int group )
116 {
117  double mean, child_net_adults, elderly_net_adults, x;
118 
119  indiv->age_group = group;
120 
121  mean = params->mean_random_interactions[group];
122  indiv->base_random_interactions = negative_binomial_draw( mean, mean );
123  update_random_interactions( indiv, params );
124 
125  if( group == AGE_18_64 )
126  {
127  child_net_adults = params->child_network_adults * params->population[AGE_0_17] / params->population[AGE_18_64];
128  elderly_net_adults = params->elderly_network_adults * params->population[AGE_65] / params->population[AGE_18_64];
129 
130  x = gsl_rng_uniform( rng );
131  if( x < child_net_adults )
132  indiv->work_network = AGE_0_17;
133  else if( x < ( elderly_net_adults + child_net_adults ) )
134  indiv->work_network = AGE_65;
135  else
136  indiv->work_network = AGE_18_64;
137  }
138  else
139  indiv->work_network = group;
140 }

142 /*****************************************************************************************
143 * Name: update_random_interactions
144 * Description: update the number of random interactions for an individual after a
145 * change in status both at the individual level or national policy
146 * Returns: void
147 ******************************************************************************************/
148 void update_random_interactions( individual *indiv, parameters* params )
149 {
150  double n = indiv->base_random_interactions;
151 
152 
153  if( !indiv->quarantined )
154  {
155  switch( indiv->status )
156  {
157  case DEATH: n = 0; break;
158  case HOSPITALISED: n = params->hospitalised_daily_interactions; break;
159  case CRITICAL: n = params->hospitalised_daily_interactions; break;
160  default: n = ifelse( params->social_distancing_on, n * params->social_distancing_random_network_multiplier, n );
161  }
162  }
163  else
164  n = params->quarantined_daily_interactions;
165 
166  indiv->random_interactions = round_random( n );
167 }
