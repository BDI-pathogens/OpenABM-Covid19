42 * Arguments: network - pointer to network which is constructed

44 * k - mean number of connections (long)

50 void build_watts_strogatz_network(

53  long k,
54  double p_rewire,
55  int randomise_nodes
56 )
57 {
58  long incr = k/2, neighbour, i, j, l;
59 
60  // Allocate memory (needed for large N)
61  long** edge_mat;
62  edge_mat = calloc(N, sizeof(long *));
63  for(i = 0; i < N; i++)
64  edge_mat[i] = calloc(k*10, sizeof(long));
65 
66  // Degree for each individual
67  int* n_edges_arr = calloc(N, sizeof(int));
68  for(i = 0; i < N; i++){
69  n_edges_arr[i] = k;
70  }
71 
72  // Step 1: Set up random lattice
73  for(i = 0; i < N; i++){
74  j = 0; l = 0;
75  while(l < k){
76  // Make sure we loop to the start of the ring
77  neighbour = (i - incr + j + N) % N;
78 
79  if(neighbour != i){
80  edge_mat[i][l] = neighbour;
81  l++;
82  }
83  j++;
84  }
85  }
86 
87  double u;
88  long new_contact, old_contact;
89 
90  // Step 2: Randomly rewire connections with probability "p_rewire"
91  for(i = 0; i < N; i++){
92  for(j = 0; j < n_edges_arr[i]; j++){
93 
94  u = gsl_rng_uniform(rng);
95 
96  if(u < p_rewire){
97 
98  // Draw a new contact (long between 0 and N-1)
99  new_contact = (long) floor(gsl_rng_uniform(rng) * N);
100 
101  // Check if new_connection is already connected (or is self)
102  while(check_member_or_self(new_contact, i, edge_mat[i], n_edges_arr[i])){
103  new_contact = gsl_rng_uniform_int(rng, N);
104  }
105 
106  // Remove contact between person "i" and the original contact
107  old_contact = edge_mat[i][j];
108 
109  remove_contact(edge_mat[old_contact], i, &(n_edges_arr[old_contact]));
110 
111  // if not, rewire to this new contact
112  edge_mat[i][j] = new_contact;
113 
114  // Add i to list of connections to new_contact (and increase number of edges)
115  add_contact(edge_mat[new_contact], i, &(n_edges_arr[new_contact]));
116  }
117  }
118  }
119  // Count total edges (i.e. network->n_edges) in undirected graph
120  // (above code constructs a directed graph)
121  long n_edges = 0;
122  for(i = 0; i < N; i++){
123  n_edges += n_edges_arr[i];
124  }
125  network->n_edges = n_edges/2;
126 
127  // Form array of total edges (i.e. network->edges)
128  network->edges = calloc(n_edges, sizeof(edge));
129 
130  // randomise the order nodes are put on the lattice if appropriate
131  long* node_list = calloc(N, sizeof(long));
132  for( i = 0; i < N; i++ )
133  node_list[i] = i;
134  if( randomise_nodes )
135  gsl_ran_shuffle( rng, node_list, N, sizeof(long) );
136 
137  long idx = 0;
138  for(i = 0; i < N; i++){
139  for(j = 0; j < n_edges_arr[i]; j++){
140  network->edges[idx].id1 = node_list[i];
141  network->edges[idx].id2 = node_list[edge_mat[i][j]];
142  idx++;
143 
144  // Remove connections between id2 -> id1
145  // (otherwise we count edges twice)
146  remove_contact(edge_mat[node_list[edge_mat[i][j]]],
147  node_list[i], &(n_edges_arr[node_list[edge_mat[i][j]]]));
148  }
149  }
150 
151  for(i = 0; i < N; i++)
152  free(edge_mat[i]);
153  free(edge_mat);
154  free(n_edges_arr);
155  free(node_list);
156 };
