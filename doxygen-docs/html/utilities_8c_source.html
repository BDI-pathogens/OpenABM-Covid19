63 /*****************************************************************************************
64 * Name: bernoulli_draw_list
65 * Description: generates a draw list so that we can efficiently sample
66 * from a distribution
67 * the 2 possible outcomes are floor(mean) and floor(mean)+1
68 *
69 * Arguments: list: pointer to draw list be filled in

73 void bernoulli_draw_list(

83  b = a + 1;

90 /*****************************************************************************************
91 * Name: geometric_max_draw_list
92 * Description: generates a draw list which is geometrically distributed with parameter
93 * p and a maximum. Note, since we allow the possibility of immediate
94 * drop out then the the first event is 0 not 1 (so if no max we have
95 * mean of 1/p-1).
96 * Any draws larger than max are given the value max
97 *
98 * Arguments: list: pointer to draw list be filled in
99 * n: length of draw list
100 * mean: mean of distribution
101 ******************************************************************************************/
102 void geometric_max_draw_list(
103  int *list,
104  int n,
105  double p,
106  int max
107 )
108 {
109  int idx;
110  int day = 0;
111  double cprob = 0;
112  double prob = p;
113  int limit = round( prob * n );
114 
115  for( idx = 0; idx < n; idx++ )
116  {
117  if( idx < limit )
118  list[idx] = day;
119  else
120  {
121  day++;
122  if( day < max )
123  {
124  cprob += prob;
125  prob *= ( 1 - p );
126  limit = round( ( cprob + prob ) * n );
127  }
128  else
129  limit = n;
130  list[idx] = day;
131  }
132  }
133 }

135 /*****************************************************************************************
136 * Name: gamma_rate_curve
137 * Description: generates a rate curve for how infectious people are based
138 * upon a discrete gamma distribution and a multiplier
139 *
140 * Arguments: list: pointer to draw list be filled in

146 void gamma_rate_curve(
147  double *list,

149  double mean,
150  double sd,
151  double factor
152 )
153 {
154  int idx = 0;
155  double a, b, total;
156

158  a = mean / b;
159 
160  total = 0;
161  for( idx = 0; idx < n; idx++ )
162  {
163  list[idx] = gsl_cdf_gamma_P( ( idx + 1 ) * 1.0, a, b ) - total;
