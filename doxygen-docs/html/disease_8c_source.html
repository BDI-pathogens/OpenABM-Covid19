55 /*****************************************************************************************
56 * Name: estimate_mean_interactions_by_age
57 * Description: estimates the weighted mean number of interactions by age
58 * each interaction is weighted by the type factor
59 * Returns: void
60 ******************************************************************************************/
61 double estimate_mean_interactions_by_age( model *model, int age )
62 {
63  long pdx, ndx;
64  long people = 0;
65  double inter = 0;
66  double *weight = model->params->relative_transmission_by_type;
67 
68  for( pdx = 0; pdx < model->params->n_total; pdx++ )
69  if( model->population[pdx].age_group == age )
70  {
71  people++;
72  inter += model->population[pdx].base_random_interactions * weight[RANDOM];
73  }
74  for( pdx = 0; pdx < model->household_network->n_edges; pdx++ )
75  {
76  if( model->population[model->household_network->edges[pdx].id1].age_group == age )
77  inter += weight[HOUSEHOLD];
78  if( model->population[model->household_network->edges[pdx].id2].age_group == age )
79  inter += weight[HOUSEHOLD];
80  }
81 
82  for( ndx = 0; ndx < N_WORK_NETWORKS ; ndx++ )
83  for( pdx = 0; pdx < model->work_network[ndx]->n_edges; pdx++ )
84  {
85  if( model->population[model->work_network[ndx]->edges[pdx].id1].age_group == age )
86  inter += model->params->daily_fraction_work * weight[WORK];
87  if( model->population[model->work_network[ndx]->edges[pdx].id2].age_group == age )
88  inter += model->params->daily_fraction_work * weight[WORK];
89  }
90 
91  return 1.0 * inter / people;
92 }

94 /*****************************************************************************************
95 * Name: set_up_infectious curves
96 * Description: sets up discrete distributions and functions which are used to
97 * model events and calculates infectious rate per interaction.
98 *
99 * The infectious rate per interaction adult is the infectious rate divided
100 * by the mean number of daily interactions of an adult.
101 *
102 * Adjustments are calculated for children and elderly based upon their
103 * difference in the number of daily interactions and the relative overall
104 * suscpetibility.
105 *
106 * Returns: void
107 ******************************************************************************************/
108 void set_up_infectious_curves( model *model )
109 {
110  parameters *params = model->params;
111  double infectious_rate, type_factor;
112  double mean_interactions[N_AGE_GROUPS];
113  int type;
114 
115  mean_interactions[AGE_0_17] = estimate_mean_interactions_by_age( model, AGE_0_17 );
116  mean_interactions[AGE_18_64] = estimate_mean_interactions_by_age( model, AGE_18_64 );
117  mean_interactions[AGE_65] = estimate_mean_interactions_by_age( model, AGE_65 );
118 
119  infectious_rate = params->infectious_rate / mean_interactions[AGE_18_64];
120  params->adjusted_susceptibility_child = params->relative_susceptibility_child * mean_interactions[AGE_18_64] / mean_interactions[AGE_0_17];
121  params->adjusted_susceptibility_elderly = params->relative_susceptibility_elderly * mean_interactions[AGE_18_64] / mean_interactions[AGE_65];
122 
123  for( type = 0; type < N_INTERACTION_TYPES; type++ )
124  {
125  type_factor = params->relative_transmission_by_type_used[type];
126 
127  gamma_rate_curve( model->event_lists[PRESYMPTOMATIC].infectious_curve[type], MAX_INFECTIOUS_PERIOD, params->mean_infectious_period,
128  params->sd_infectious_period, infectious_rate * type_factor );
129 
130  gamma_rate_curve( model->event_lists[ASYMPTOMATIC].infectious_curve[type], MAX_INFECTIOUS_PERIOD, params->mean_infectious_period,
131  params->sd_infectious_period, infectious_rate * type_factor * params->asymptomatic_infectious_factor);
132 
133  gamma_rate_curve( model->event_lists[SYMPTOMATIC].infectious_curve[type], MAX_INFECTIOUS_PERIOD, params->mean_infectious_period,
134  params->sd_infectious_period, infectious_rate * type_factor );
135 
136  gamma_rate_curve( model->event_lists[HOSPITALISED].infectious_curve[type], MAX_INFECTIOUS_PERIOD, params->mean_infectious_period,
137  params->sd_infectious_period, infectious_rate * type_factor );
138 
139  gamma_rate_curve( model->event_lists[CRITICAL].infectious_curve[type], MAX_INFECTIOUS_PERIOD, params->mean_infectious_period,
140  params->sd_infectious_period, infectious_rate * type_factor );
141  };
142 }
143 /*****************************************************************************************
144 * Name: transmit_virus_by_type
145 * Description: Transmits virus over the interaction network for a type of
146 * infected people
147 * Returns: void
148 ******************************************************************************************/
149 void transmit_virus_by_type(
150  model *model,
151  int type
152 )
153 {
154  long idx, jdx, n_infected;
155  int day, n_interaction;
156  double hazard_rate;
157  event_list *list = &(model->event_lists[type]);
158  event *event, *next_event;
159  interaction *interaction;
160  individual *infector;
161 
162  for( day = model->time-1; day >= max( 0, model->time - MAX_INFECTIOUS_PERIOD ); day-- )
163  {
164  n_infected = list->n_daily_current[ day];
165  next_event = list->events[ day ];
166 
167  for( idx = 0; idx < n_infected; idx++ )
168  {
169  event = next_event;
170  next_event = event->next;
171 
172  infector = event->individual;
173  n_interaction = infector->n_interactions[ model->interaction_day_idx ];
174  if( n_interaction > 0 )
175  {
176  interaction = infector->interactions[ model->interaction_day_idx ];
177  hazard_rate = list->infectious_curve[interaction->type][ model->time - 1 - time_infected( infector) ];
178 
179  for( jdx = 0; jdx < n_interaction; jdx++ )
180  {
181  if( interaction->individual->status == UNINFECTED )
182  {
183  interaction->individual->hazard -= hazard_rate;
184  if( interaction->individual->hazard < 0 )
185  new_infection( model, interaction->individual, infector );
186  }
187  interaction = interaction->next;
188  }
189  }
190  }
191  }
192 }

214 /*****************************************************************************************
215 * Name: new_infection
216 * Description: infects a new individual
217 * Returns: void
218 ******************************************************************************************/
219 void new_infection(
220  model *model,
221  individual *infected,
222  individual *infector
223 )
224 {
225  infected->infector = infector;
226 
227  if( gsl_ran_bernoulli( rng, model->params->fraction_asymptomatic ) )
228  {
229  transition_one_disese_event( model, infected, NO_EVENT, ASYMPTOMATIC, NO_EDGE );
230  transition_one_disese_event( model, infected, ASYMPTOMATIC, RECOVERED, ASYMPTOMATIC_RECOVERED );
231  }
232  else
233  {
234  transition_one_disese_event( model, infected, NO_EVENT, PRESYMPTOMATIC, NO_EDGE );
235  transition_one_disese_event( model, infected, PRESYMPTOMATIC, SYMPTOMATIC, PRESYMPTOMATIC_SYMPTOMATIC );
236  }

240 * Name: transition_one_disease_event
241 * Description: Generic function for updating an individual with their next
242 * event and adding the applicable events

245 void transition_one_disese_event(
246  model *model,
247  individual *indiv,
248  int from,
249  int to,
250  int edge
251 )
252 {
253  indiv->status = from;
254 
255  if( from != NO_EVENT )
256  indiv->time_event[from] = model->time;
257  if( indiv->current_disease_event != NULL )
258  remove_event_from_event_list( model, indiv->current_disease_event );
259  if( indiv->next_disease_event != NULL )
260  indiv->current_disease_event = indiv->next_disease_event;
261 
262  if( to != NO_EVENT )
263  {
264  indiv->time_event[to] = model->time + ifelse( edge == NO_EDGE, 0, sample_transition_time( model, edge ) );
265  indiv->next_disease_event = add_individual_to_event_list( model, to, indiv, indiv->time_event[to] );
266  }

269 /*****************************************************************************************
270 * Name: transition_to_symptomatic
271 * Description: Transitions infected who are due to become symptomatic. At this point
272 * there are 2 choices to be made:
273 *
274 * 1. Will the individual require hospital treatment or will
275 * they recover without needing treatment
276 * 2. Does the individual self-quarantine at this point and
277 * asks for a test
278 *
279 * Returns: void
280 ******************************************************************************************/
281 void transition_to_symptomatic( model *model, individual *indiv )
282 {
283  if( gsl_ran_bernoulli( rng, model->params->hospitalised_fraction[ indiv->age_group ] ) )
284  transition_one_disese_event( model, indiv, SYMPTOMATIC, HOSPITALISED, SYMPTOMATIC_HOSPITALISED );
285  else
286  transition_one_disese_event( model, indiv, SYMPTOMATIC, RECOVERED, SYMPTOMATIC_RECOVERED );
287 
288  intervention_on_symptoms( model, indiv );
289 }
