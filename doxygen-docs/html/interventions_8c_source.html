52 /*****************************************************************************************
53 * Name: update_intervention_policy
54 * Description: Updates the intervention policy by adjusting parmaters
55 ******************************************************************************************/
56 void update_intervention_policy( model *model, int time )
57 {
58  parameters *params = model->params;
59  long pdx;
60  int type;
61 
62  params->app_turned_on = ( time >= params->app_turn_on_time );
63 
64  if( time == 0 )
65  {
66  params->social_distancing_on = FALSE;
67  params->daily_fraction_work_used = params->daily_fraction_work;
68  for( type = 0; type < N_INTERACTION_TYPES; type++ )
69  params->relative_transmission_by_type_used[type] = params->relative_transmission_by_type[type];
70  }
71  else
72  if( time == params->social_distancing_time_on )
73  {
74  params->social_distancing_on = TRUE;
75  params->daily_fraction_work_used = params->daily_fraction_work * params->social_distancing_work_network_multiplier;
76 
77  params->relative_transmission_by_type_used[HOUSEHOLD] = params->relative_transmission_by_type[HOUSEHOLD] *
78  params->social_distancing_house_interaction_multiplier;
79  set_up_infectious_curves( model );
80 
81  for( pdx = 0; pdx < params->n_total; pdx++ )
82  update_random_interactions( &(model->population[pdx]), params );
83  }
84  else
85  if( time == params->social_distancing_time_off )
86  {
87  params->social_distancing_on = FALSE;
88  params->daily_fraction_work_used = params->daily_fraction_work;
89 
90  params->relative_transmission_by_type_used[HOUSEHOLD] = params->relative_transmission_by_type[HOUSEHOLD];
91  set_up_infectious_curves( model );
92 
93  for( pdx = 0; pdx < params->n_total; pdx++ )
94  update_random_interactions( &(model->population[pdx]), params );
95  }
96 };

105 {

114

118  return;

157 /*****************************************************************************************
158 * Name: intervention_test_take
159 * Description: An individual takes a test

163 * Returns: void

165 void intervention_test_take( model *model, individual *indiv )
166 {

175  }

180 /*****************************************************************************************
181 * Name: intervention_test_result
182 * Description: An individual gets a test result
183 *
184 * 1. On a negative result the person is released from quarantine
185 * 2. On a positive result they become a case and trigger the
186 * intervention_on_positive_result cascade
187 * Returns: void
188 ******************************************************************************************/
189 void intervention_test_result( model *model, individual *indiv )
190 {
191  if( indiv->quarantine_test_result == FALSE )
192  {
193  if( indiv->quarantined )
194  intervention_quarantine_release( model, indiv );
195  }
196  else
197  {
198  if( indiv->is_case == FALSE )
199  {
200  set_case( indiv, model->time );
201  add_individual_to_event_list( model, CASE, indiv, model->time );
202  }
203 
204  if( !in_hospital( indiv ) || !(model->params->allow_clinical_diagnosis) )
205  intervention_on_positive_result( model, indiv );
206  }
207  indiv->quarantine_test_result = NO_TEST;

210 /*****************************************************************************************
211 * Name: intervention_notify_contracts
212 * Description: If the individual is an app user then we loop over the stored contacts
213 * and notifies them.
214 * Start from the oldest date, so that if we have met a contact
215 * multiple times we order the test from the first contact
216 * Returns: void
217 ******************************************************************************************/
218 void intervention_notify_contacts(
219  model *model,
220  individual *indiv,
221  int level
222 )
223 {
224  if( !indiv->app_user || !model->params->app_turned_on )
225  return;
226 
227  interaction *inter;
228  individual *contact;
229  parameters *params = model->params;
230  int idx, ddx, day, n_contacts;
231 
232  day = model->interaction_day_idx;
233  for( ddx = 0; ddx < params->quarantine_days; ddx++ )
234  ring_dec( day, model->params->days_of_interactions );
235 
236  for( ddx = params->quarantine_days - 1; ddx >=0; ddx-- )
237  {
238  n_contacts = indiv->n_interactions[day];
239 
240  if( n_contacts > 0 )
241  {
242  inter = indiv->interactions[day];
243  for( idx = 0; idx < n_contacts; idx++ )
244  {
245  contact = inter->individual;
246  if( contact->app_user )
247  {
248  if( inter->traceable == UNKNOWN )
249  inter->traceable = gsl_ran_bernoulli( rng, params->traceable_interaction_fraction );
250  if( inter->traceable )
251  intervention_on_traced( model, contact, model->time - ddx, level + 1 );
252  }
253  inter = inter->next;
254  }
255  }
256  ring_inc( day, model->params->days_of_interactions );
257  }
258 }

268  int time,

278  time_event = ifelse( time != UNKNOWN, time, model->time + sample_transition_time( model, TRACED_QUARANTINE ) );

285

287  intervention_notify_contacts( model, contact, NOT_RECURSIVE );

291 /*****************************************************************************************
292 * Name: intervention_on_symptoms
293 * Description: The interventions performed upon showing symptoms of a flu-like symptoms
294 *
295 * 1. If in quarantine already or drawn for self-quarantine then
296 * quarantine for length of time symptomatic people do
297 * 2. If we have community testing on symptoms and a test has not been
298 * ordered already then order one
299 * 3. Option to quarantine all household members upon symptoms
300 * Returns: void
301 ******************************************************************************************/
302 void intervention_on_symptoms( model *model, individual *indiv )
303 {
304  int quarantine, time_event;
305 
306  quarantine = indiv->quarantined || gsl_ran_bernoulli( rng, model->params->self_quarantine_fraction );
307 
308  if( quarantine )
309  {
310  time_event = model->time + sample_transition_time( model, SYMPTOMATIC_QUARANTINE );
311  intervention_quarantine_until( model, indiv, time_event, TRUE );
312 
313  if( model->params->quarantine_household_on_symptoms )
314  intervention_quarantine_household( model, indiv, time_event, FALSE );
315 
316  if( model->params->test_on_symptoms )
317  intervention_test_order( model, indiv, model->time + model->params->test_order_wait );
318  }
319 }

374 /*****************************************************************************************
375 * Name: intervention_on_traced
376 * Description: Optional interventions performed upon becoming contact-traced
377 *
378 * 1. Quarantine the individual
379 * 2. Quarantine the individual and their household
380 * 2. Order a test for the individual
381 * 4. Recursive contact-trace
382 *
383 * Arguments: model - pointer to model
384 * indiv - pointer to person being traced
385 * contact_time - time at which the contact was made
386 * recursion level - layers of the network to reach this connection
387 *
388 * Returns: void
389 ******************************************************************************************/
390 void intervention_on_traced(
391  model *model,
392  individual *indiv,
393  int contact_time,
394  int recursion_level
395 )
396 {
397  if( in_hospital( indiv ) || indiv->is_case )
398  return;
399 
400  parameters *params = model->params;
401 
402  if( params->quarantine_on_traced )
403  {
404  int time_event = model->time + sample_transition_time( model, TRACED_QUARANTINE );
405  intervention_quarantine_until( model, indiv, time_event, TRUE );
406 
407  if( params->quarantine_household_on_traced )
408  intervention_quarantine_household( model, indiv, time_event, FALSE );
409  }
410 
411  if( params->test_on_traced )
412  {
413  int time_test = max( model->time + params->test_order_wait, contact_time + params->test_insensititve_period );
414  intervention_test_order( model, indiv, time_test );
415  }
416 
417  if( recursion_level < params->tracing_network_depth )
418  intervention_notify_contacts( model, indiv, recursion_level + 1 );
419 }
