307 /*****************************************************************************************
308 * Name: calculate_household_distribution
309 * Description: Calculates the number of households of each size from the UK
310 * household survey data. The age split of the households is based
311 * upon ONS data for each age group. Then:
312 *
313 * 1. All elderly are assumed to live in 1 or 2 person household.
314 * We fill them up in equal proportion
315 * 2. The same proportion of household with 3/4/5/6 people in have
316 * children and then n-2 of the occupents are assumed to be children.
317 *
318 * Note there is no age mixing in household between elderly and others.
319 *
320 * Argument: model - pointer to the model
321 * n_house_tot - number of households which this function sets
322 * elderly_frac_1_2 - fraction of 1/2 person households which are elderly
323 * child_frac_2_6, - fraction of 3/4/5/6 person households which have chlildren
324 ******************************************************************************************/
325 void calculate_household_distribution(
326  model *model,
327  long *n_house_tot,
328  double *elderly_frac_1_2,
329  double *child_frac_3_6
330 )
331 {
332  long total;
333  int idx;
334  double survey_tot, max_children, pop_all;
335  double n_person_frac[ HOUSEHOLD_N_MAX ];
336  double *pop = model->params->population;
337 
338  pop_all = pop[AGE_0_17] + pop[AGE_18_64] + pop[AGE_65];
339  survey_tot = 0;
340  max_children = 0;
341  for( idx = 0; idx < HOUSEHOLD_N_MAX; idx++)
342  {
343  n_person_frac[idx] = model->params->household_size[ idx ] * ( idx + 1 );
344  survey_tot += n_person_frac[idx];
345  max_children += model->params->household_size[ idx ] * max( idx -1 , 0 );
346  }
347 
348  *child_frac_3_6 = pop[AGE_0_17] / pop_all / ( max_children / survey_tot );
349  *elderly_frac_1_2 = pop[AGE_65] / pop_all / ( ( n_person_frac[HH_1] + n_person_frac[HH_2] ) / survey_tot );
350 
351  if( *child_frac_3_6 > 1 )
352  print_exit( "not sufficient 3-6 person households for all the children" );
353 
354  if( *elderly_frac_1_2 > 1 )
355  print_exit( "not sufficient 1-2 person households for all the elderly" );
356 
357  total = 0;
358  for( idx = 1; idx < HOUSEHOLD_N_MAX; idx++)
359  {
360  n_house_tot[idx] = (long) round( n_person_frac[idx] / survey_tot / ( idx + 1 ) * model->params->n_total );
361  total += n_house_tot[idx] * ( idx + 1 );
362  }
363  n_house_tot[0] = model->params->n_total - total;
364 }

366 /*****************************************************************************************
367 * Name: build_household_network
368 * Description: Builds a network of household interactions based upon the UK household
369 * data. Note this can only be done once since it allocates new memory
370 * to the network structure.
371 *
372 * As part of the household allocation process we also assign people
373 * to age groups. All elderly are assumed to live in 1-2 person
374 * households and all children are assumed to live in a house with
375 * 2 adults.
376 ******************************************************************************************/
377 void build_household_network( model *model )
378 {
379  long pop_idx, hdx, edge_idx, house_no;
380  int ndx, pdx, p2dx, age;
381  long n_house_tot[ HOUSEHOLD_N_MAX ];
382  double elderly_frac_1_2;
383  double child_frac_3_6;
384  network *network = model->household_network;
385 
386  if( network->n_edges != 0 )
387  print_exit( "the household network can only be once" );
388 
389  calculate_household_distribution( model, n_house_tot, &elderly_frac_1_2, &child_frac_3_6 );
390  for( ndx = 0; ndx < HOUSEHOLD_N_MAX; ndx++ )
391  network->n_edges += n_house_tot[ndx] * ndx * ( ndx + 1 ) / 2;
392  network->edges = calloc( network->n_edges, sizeof( edge ) );
393 
394  model->household_directory = calloc( 1, sizeof( var_array ) );
395  model->household_directory->n_idx = 0;
396  for( ndx = 0; ndx < HOUSEHOLD_N_MAX; ndx++ )
397  model->household_directory->n_idx += n_house_tot[ndx];
398  model->household_directory->n_jdx = calloc( model->household_directory->n_idx, sizeof( int ) );
399  model->household_directory->val = calloc( model->household_directory->n_idx, sizeof( long* ) );
400 
401  edge_idx = 0;
402  pop_idx = 0;
403  house_no = 0;
404  for( ndx = 0; ndx < HOUSEHOLD_N_MAX; ndx++ )
405  for( hdx = 0; hdx < n_house_tot[ndx]; hdx++ )
406  {
407  model->household_directory->n_jdx[house_no] = ndx + 1;
408  model->household_directory->val[house_no] = calloc( ndx + 1, sizeof( long ) );
409  for( pdx = 0; pdx < ( ndx + 1 ); pdx++ )
410  {
411  age = AGE_18_64;
412  if( ndx <= HH_2 && ( 1.0 * hdx / n_house_tot[ndx] ) < elderly_frac_1_2 )
413  age = AGE_65;
414  if( ndx >= HH_3 && ( 1.0 * hdx / n_house_tot[ndx] ) < child_frac_3_6 && pdx < ( ndx - 1 ) )
415  age = AGE_0_17;
416 
417  set_age_group( &(model->population[pop_idx]), model->params, age );
418  set_house_no( &(model->population[pop_idx]), house_no );
419  model->household_directory->val[house_no][pdx] = pop_idx;
420 
421  for( p2dx = pdx+1; p2dx < ( ndx + 1 ); p2dx++ )
422  {
423  network->edges[edge_idx].id1 = pop_idx;
424  network->edges[edge_idx].id2 = pop_idx + p2dx - pdx;
425 
426  edge_idx++;
427  }
428  pop_idx++;
429  }
430  house_no++;
431  }
432 }

608 /*****************************************************************************************
609 * Name: build_random_newtork
610 * Description: Builds a new random network
611 ******************************************************************************************/
612 void build_random_network( model *model )
613 {
614  long idx, n_pos, person;
615  int jdx;
616  long *interactions = model->possible_interactions;
617  network *network = model->random_network;
618 
619  network->n_edges = 0;
620  n_pos = 0;
621  for( person = 0; person < model->params->n_total; person++ )
622  for( jdx = 0; jdx < model->population[person].random_interactions; jdx++ )
623  interactions[n_pos++]=person;
624 
625  if( n_pos == 0 )
626  return;
627 
628  gsl_ran_shuffle( rng, interactions, n_pos, sizeof(long) );
629 
630  idx = 0;
631  n_pos--;
632  while( idx < n_pos )
633  {
634  if( interactions[ idx ] == interactions[ idx + 1 ] )
635  {
636  idx++;
637  continue;
638  }
639  network->edges[network->n_edges].id1 = interactions[ idx++ ];

649 void add_interactions_from_network(
650 
651  model *model,
652  network *network,
653  int skip_hospitalised,
654  int skip_quarantined,
655  double prob_drop
656 )
657 {

660  int day = model->interaction_day_idx;
661 
662  interaction *inter1, *inter2;
663  individual *indiv1, *indiv2;
664 
665  while( idx < network->n_edges )
666  {

668  indiv2 = &(model->population[ network->edges[idx++].id2 ] );

673  continue;
674  if( skip_quarantined && ( indiv1->quarantined || indiv2->quarantined ) )
675  continue;
676  if( prob_drop > 0 && gsl_ran_bernoulli( rng, prob_drop ) )
677  continue;
678 
679  inter1 = &(model->interactions[ all_idx++ ]);
680  inter2 = &(model->interactions[ all_idx++ ]);
681 
682  inter1->type = network->type;
683  inter1->traceable = UNKNOWN;
684  inter1->individual = indiv2;
685  inter1->next = indiv1->interactions[ day ];
686  indiv1->interactions[ day ] = inter1;
687  indiv1->n_interactions[ day ]++;
688 
689  inter2->type = network->type;
690  inter2->traceable = UNKNOWN;
691  inter2->individual = indiv1;
692  inter2->next = indiv2->interactions[ day ];
693  indiv2->interactions[ day ] = inter2;
694  indiv2->n_interactions[ day ]++;
695 
696  model->n_total_intereactions++;
697 
698  if( all_idx > model->n_interactions )
699  all_idx = 0;
700 
701  }
